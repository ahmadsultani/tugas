program ArrayBaru
{Membaca dari keyboard 3 array A, B, C masing masing 5 elemen kemudian mengisikan A, B, C ke matriks D 3x5. Kemudian rata-rata dari matriks D dihitung. Di akhir tampilkan array A, B, C, D dan rata}

DEKLARASI
RATA : real
data : integer
n, row, col : integer
A, B, C : array [1..n] of integer
D : array [1..3, 1..n] of integer
procedure ReadArr(input arr, size : integer)
{mengisi elemen array dengan membaca data dari keyboard}
procedure PrintArr(input arr, size : integer)
{menampilkan elemen array}

ALGORITMA
n <- 5
RATA <- 0
ReadArr(A, n)
ReadArr(B, n)
ReadArr(C, n)
for row <- 1 to 3 do
    for col <- 1 to n do
        case row of
        1: data <- A[col]
        2: data <- B[col]
        3: data <- C[col]
        endcase
    D[row, col] <- data
    RATA <- RATA + data
endfor
PrintArr(A, n)
PrintArr(B, n)
PrintArr(C, n)
for row <- 1 to 3 do
    for col <- 1 to n do
        write(D[row, col])
    endfor
    write("\n")
endfor
RATA <- RATA / (3 * n)
write(RATA)
